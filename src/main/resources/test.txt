@brojElixira@ saberi(brojElixira a, brojElixira b) #
    krajBorbe a + b;
$

@imeKartice@ izaberiIme(brojElixira v) #
    leader(v > 0) #
        krajBorbe "Musketeer";
    $
    member #
        krajBorbe "Pekka";
    $
$

@bezElixira@ prikazi(imeKartice msg) #
    ispisiKarticu(msg);
    krajBorbe;
$

battle() #
    brojElixira x = 7;
    brojElixira y = 5;

    brojElixira z = saberi(x, y);
    imeKartice t = izaberiIme(z);
    prikazi(t);

    krajBorbe;
$
